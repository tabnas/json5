# Shared list of json5/json5-tests fixtures both plugin ports skip.
#
# Each implementation leans on its host Jsonic library (TypeScript or Go),
# and each host has a handful of corners where it is more permissive than
# the JSON5 spec, or slightly stricter. Rather than hide this asymmetry
# in separate lists, the union is kept here and shared by both suites so
# the two implementations exhibit IDENTICAL pass/fail behaviour on the
# official corpus.
#
# A fixture is listed here if EITHER implementation fails it.
#
# Format: one fixture-relative path per non-empty, non-comment line.

# Jsonic accepts a file containing only comments/whitespace as empty.
comments/top-level-inline-comment.txt
comments/top-level-block-comment.txt

# Jsonic's whitespace set does not include every Unicode Zs category char.
misc/valid-whitespace.json5

# Backslash + CRLF line continuation in strings is not handled.
new-lines/escaped-crlf.json5

# Jsonic TS's number matcher does not accept the `0X` uppercase hex
# prefix (Go accepts it). Skipped to match TS behaviour.
numbers/hexadecimal-uppercase-x.json5

# Jsonic Go's number matcher does not accept a trailing decimal point
# immediately followed by an exponent ("5.e4"). Skipped to match Go.
numbers/float-trailing-decimal-point-with-integer-exponent.json5

# Jsonic accepts JavaScript octal and noctal number literals.
numbers/negative-noctal.js
numbers/negative-octal.txt
numbers/negative-zero-octal.txt
numbers/noctal-with-leading-octal-digit.js
numbers/noctal.js
numbers/octal.txt
numbers/positive-noctal.js
numbers/positive-octal.txt
numbers/positive-zero-octal.txt
numbers/zero-octal.txt

# Jsonic accepts malformed object keys and lone trailing commas.
objects/illegal-unquoted-key-number.txt
objects/illegal-unquoted-key-symbol.txt
objects/lone-trailing-comma-object.txt
